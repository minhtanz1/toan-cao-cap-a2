\section{\S 6: Các phép biến sơ cấp}
\subsection{Phép biến đổi sơ cấp gồm 3 loại:}
\begin{enumerate}
	\item Đổi vị trí 2 dòng của ma trận
	\item Nhân 1 dòng với một số $\neq 0$
	\item Lấy dòng này + hoặc - với k lần dòng khác.
\end{enumerate}
\subsection{Ứng dụng}
\subsubsection{Ứng dụng tìm ma trận nghịch đảo $A^{-1}$}
$ (A_{n \times n} | I_n) \rightarrow \cdots \rightarrow (I | A^{-1})$
\subsubsection{Ứng dụng tìm hạng (Rank) ma trận}
\begin{enumerate}
	\item Ma trận có dạng \textbf{Bậc thang.} \\
